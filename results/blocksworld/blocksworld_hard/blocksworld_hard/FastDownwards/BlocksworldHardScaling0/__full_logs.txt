    (:requirements :strips :typing)
    
    (:types

    )
    
    (:predicates

        (ontable ?x - block)        ; X is resting directly on the table

            (clear ?x) 
            (handempty)
        )
        :effect (and 
            (not (ontable ?x))
            (not (clear ?x))
            (not (handempty))
            (holding ?x)
        )

        :parameters (?x - block)
        :precondition (holding ?x)

            (clear ?x) 
            (handempty)
            (not (holding ?x))
