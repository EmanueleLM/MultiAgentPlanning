  )

    (close-stephanie) - number

  )

  (:action wait-1
    :parameters ()
    :precondition (and)
    :effect (increase (now) 1)
  )

  (:action travel
    :parameters (?from - location ?to - location)
    :precondition (at ?from)
    :effect (and

  )

      (< (now) (close-stephanie))
    )

      (increase (now) 1)
      (increase (met-stephanie) 1)
    )
  )
)
