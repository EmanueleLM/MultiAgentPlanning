(define (domain fastdownwards-schedule)
  (:requirements :strips :typing :negative-preconditions :equality)
  (:types person slot report)
  (:constants alice bob carol - person)
  (:predicates
    (reported-available ?r - report ?p - person ?s - slot) ; raw report facts
    (free ?p - person ?s - slot)                          ; agent free at slot (set by agent-specific actions)
    (all-free ?s - slot)                                  ; all three free at slot (can be derived via action derive-all-free)
    (earliest-allowed ?s - slot)                          ; precomputed earliest slot(s) satisfying tie-break constraints
    (meeting-scheduled)                                   ; goal achieved
    (meeting-start ?s - slot)                             ; meeting starts at slot
    (bob-morning ?s - slot)                               ; marks Bob's morning slots (to be set from Bob's report)
  )

  ;; Agent-specific availability actions (distinct and identifiable by agent per requirement)
  (:action mark-free-alice
    :parameters (?r - report ?s - slot)
    :precondition (reported-available ?r alice ?s)
    :effect (free alice ?s)
  )

  (:action mark-free-bob
    :parameters (?r - report ?s - slot)
    :precondition (reported-available ?r bob ?s)
    :effect (free bob ?s)
  )

  (:action mark-free-carol
    :parameters (?r - report ?s - slot)
    :precondition (reported-available ?r carol ?s)
    :effect (free carol ?s)
  )

  (:action derive-all-free
    :parameters (?s - slot)
    :precondition (and (free alice ?s) (free bob ?s) (free carol ?s))
    :effect (all-free ?s)
  )

  ;; Scheduling action: schedule the 30-minute meeting at slot ?s.
