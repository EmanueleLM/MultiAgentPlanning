(define (domain meeting-scheduling)
  (:requirements :typing :negative-preconditions)
  (:types person slot date)
  (:predicates
    (person ?p - person)
    (slot ?s - slot)
    (date ?d - date)
    (on-date ?s - slot ?d - date)
    (free ?p - person ?s - slot)
    (preferred-slot ?s - slot)
    (scheduled ?s - slot)
  )
